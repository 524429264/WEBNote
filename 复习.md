# 复习

### 深拷贝

```js
    function clone(data) {
      var obj;
      if (data instanceof Array) {
        obj = [];
        for (var i in data) {
          obj.push(clone(data[i]))
        }
      } else if (data instanceof Object) {
        obj = {}
        for (var i in data) {
          obj[i] = clone(data[i])
        }
      } else {
        return data;
      }
      return obj
    }
```

### 防抖

```js
    function debounce(cb, delay) {
      var timer = null;
      return function () {
        var context = this;
        var args = arguments;
        clearTimeout(timer);
        timer = setTimeout(function () {
          cb.apply(context, args)
        }, delay);
      }
    }
```

### 节流

```js
    function throttle(cb, delay) {
      var begin = Date.now();
      return function () {
        var context = this;
        var args = arguments;
        var current = Date.now();
        if (current - begin >= delay) {
          cb.apply(context, args);
          begin = current;
        }
      }
    }
```
### new

```js
    function car(id) {
      this.id = id;
      this.age = 18;
    }

//new的实际过程，创建临时对象，绑定原型，return回临时对象。
    function car(id) {
      var temp = {};
      temp.__proto__ = car.prototype;
      temp.id = id;
      temp.age = 18;
      return temp
    }

    var bus = new car(30)
```
